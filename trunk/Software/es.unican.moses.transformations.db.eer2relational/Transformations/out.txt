CREATE TRIGGER Km0_Usados_exclusive ON Km0
AFTER INSERT, UPDATE
AS BEGIN

IF (SELECT count(*) FROM Usados t, inserted i
WHERE t.Cod_Coches = i.Cod_Coches) > 0
BEGIN 
RAISERROR ('The pk exists in another table', 16, 1)
ROLLBACK TRANSACTION
RETURN
END	
END
